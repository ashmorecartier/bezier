  \begin{equation*}
      B_{i}^{n}\left(t\right) = \sum_{k=0}^{n} C_{i, k}^{n} t^{k} \quad \text{avec} \quad \\
      \left \{
        \begin{array}{ll}
          C_{i, k}^{n} & = 0 \quad \text{si} \quad 0 \leq k < i \\
          C_{i, k}^{n} & = {\left(-1\right)}^{k-i} \frac{n!}{i! {\left({k-i}\right)!} {\left({n-k}\right)!}} \quad \text{si} \quad i \leq k \leq n \\
        \end{array}
      \right.
  \end{equation*}
